MODULE user_command_0100 INPUT.
   CASE sy-ucomm.
   	WHEN 'BACK'.
      leave to screen 0.
   	WHEN 'ENTER'.
   	WHEN 'SEARCH'.
*      clear: zsb002001-gesch_a,
*             zsb002001-gesch_f,
*             zsb002001-gesch_m.
*      zsb002001-gesch_m = 'X'.
      call screen 101 STARTING AT 10 10.
   ENDCASE.
ENDMODULE.

*&---------------------------------------------------------------------*

*----------------------------------------------------------------------*
MODULE exit INPUT.
  CASE sy-ucomm.
    WHEN 'EXIT'.
      leave program.
    WHEN 'CANCEL'.
      leave to screen 0.
  ENDCASE.
ENDMODULE.

*----------------------------------------------------------------------*
MODULE user_command_0101 INPUT.
  CASE ok_code.
    WHEN 'CLOSE'.
      leave to screen 0.
    WHEN 'ENTER'.
  ENDCASE.
    MESSAGE s999(zmcb99) with  sy-ucomm ',' ok_code.
ENDMODULE.
