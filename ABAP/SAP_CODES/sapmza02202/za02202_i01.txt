MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'EXIT'.
      LEAVE PROGRAM.
    WHEN 'CANCEL'.
      SET SCREEN 0.
      LEAVE SCREEN.
    WHEN 'ENTER'.

    WHEN 'S200'.
      SET SCREEN 200.

    WHEN 'C200'.
      CALL SCREEN 200.


  ENDCASE.
ENDMODULE.
