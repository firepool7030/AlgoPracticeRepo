MODULE exit INPUT.
  CASE ok_code.
    WHEN 'EXIT'.
      leave program.
    WHEN 'CANCEL'.
      leave to screen 0.
  ENDCASE.
ENDMODULE.

MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      leave to screen 0.
    WHEN 'SEARCH'.
      PERFORM search_info.
  ENDCASE.
ENDMODULE.
