MODULE exit INPUT.
  CASE sy-ucomm.
    WHEN 'EXIT'.
      LEAVE PROGRAM.
    WHEN 'CANC'.
      " 기본으로 복귀
      PERFORM set_default.
      LEAVE SCREEN.
  ENDCASE.
ENDMODULE.

MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'SEARCH'.
      " 항공사명 취득
      PERFORM get_airline_name USING zsb003301-carrid
                               CHANGING zsb003301-carrname.
      "기본 예약 List
      PERFORM get_booking_list.
    WHEN 'ENTER'.
      " 항공사명 취득
      PERFORM get_airline_name USING zsb003301-carrid
                               CHANGING zsb003301-carrname.
    WHEN OTHERS.

  ENDCASE.
ENDMODULE.
