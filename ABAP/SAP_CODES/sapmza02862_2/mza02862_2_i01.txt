MODULE exit INPUT.
  CASE sy-ucomm.
    WHEN 'EXIT'.
      LEAVE PROGRAM.
    WHEN 'CANC'.
      LEAVE TO SCREEN 0.
  ENDCASE.
ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'SEARCH'.
      PERFORM get_mat_list.

    WHEN 'ENTER'.

    WHEN 'DETAIL'.
      " 상세조회
      PERFORM get_detail_info CHANGING gv_subrc.
      IF gv_subrc IS INITIAL.
        CALL SCREEN 0200.
      ENDIF.

    WHEN OTHERS.

  ENDCASE.
ENDMODULE.
