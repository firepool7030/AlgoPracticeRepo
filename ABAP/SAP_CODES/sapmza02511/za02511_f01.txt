FORM get_info .

  select single *
    from ztb0001
    into CORRESPONDING FIELDS OF zsa022012
   where pernr = zsa022011-pernr.

  IF sy-subrc <> 0.
    message i005(zmsg_a_02) with 'Error!'.
  ENDIF.

  PERFORM get_team_info.

ENDFORM.
