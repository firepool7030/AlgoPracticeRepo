MODULE set_alv_0100 OUTPUT.
  IF go_con IS INITIAL.
    PERFORM create_alv_0100 USING 'CON_MAT' go_con go_alv.
    CREATE OBJECT go_handler.
    "set HANDLER lcl_handler=>on_double_click FOR go_alv. " => 는 사전에 정의된 메서드를 호출
    SET HANDLER go_handler->on_double_click FOR go_alv." ACTIVATION 'X'.
  ENDIF.
  PERFORM display_alv_0100.

ENDMODULE.
