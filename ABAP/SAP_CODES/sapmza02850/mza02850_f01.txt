FORM set_defualt_condition USING VALUE(p_datum) TYPE sy-datum
                           CHANGING p_begda TYPE sy-datum
                                    p_endda TYPE sy-datum.

  CLEAR: p_begda, p_endda.
  CONCATENATE sy-datum+0(4) '0101' INTO p_begda.
  p_endda = sy-datum.
ENDFORM.
