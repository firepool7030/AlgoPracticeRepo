MODULE user_command_0100 INPUT.
  CASE sy-ucomm.
    WHEN 'BACK'.
      leave to screen 0.
    WHEN 'ENTER'.
    WHEN 'SEARCH'.
      PERFORM get_info.
  ENDCASE.
ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  EXIT  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE exit INPUT.
  CASE sy-ucomm.
    WHEN 'EXIT'.
      leave program.
    WHEN 'CANCEL'.
      leave to screen 0.
  ENDCASE.
ENDMODULE.
